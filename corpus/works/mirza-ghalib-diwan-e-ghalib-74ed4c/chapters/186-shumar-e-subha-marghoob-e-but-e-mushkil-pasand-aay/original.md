شمار_سبحہ مرغوب_بت_مشکل_پسند آیا
تماشاۓ_بہ_یک_کف بردن_صد_دل_پسند آیا

بہ_فیض_بے_دلی نومیدی_جاوید آساں ہے
کشایش کو ہمارا عقدۂ_مشکل_پسند آیا

ہوائے_سیر_گل آئینۂ_بے_مہری_قاتل
کہ انداز_بہ_خوں_غلتیدن_بسمل_پسند آیا

روانی_ہاۓ_موج_خون_بسمل سے ٹپکتا ہے
کہ لطف_بے_تحاشا_رفتن_قاتل_پسند آیا

اسدؔ ہر جا سخن نے طرح_باغ_تازہ ڈالی ہے
مجھے رنگ_بہار_ایجادی_بیدل_پسند آیا

سواد چشم بسمل انتخاب نقطہ آرائی
خرام ناز بے پروائی قاتل پسند آیا

ہوئی جس کو بہار فرصت ہستی سے آگاہی
برنگ لالہ جام بادہ پر محمل پسند آیا
