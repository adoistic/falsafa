رخ_نگار سے ہے سوز_جاودانی_شمع
ہوئی ہے آتش_گل آب_زندگانی_شمع

زبان_اہل_زباں میں ہے مرگ_خاموشی
یہ بات بزم میں روشن ہوئی زبانی_شمع

کرے ہے صرف بہ_ایماۓ_شعلہ قصہ تمام
بہ_طرز_اہل_فنا ہے فسانہ_خوانی_شمع

غم اس کو حسرت_پروانہ کا ہے اے شعلے
ترے لرزنے سے ظاہر ہے نا_توانی_شمع

ترے خیال سے روح اہتزاز کرتی ہے
بہ_جلوہ_ریزی_باد_و_بہ_پر_فشانی_شمع

نشاط_داغ_غم_عشق کی بہار نہ پوچھ
شگفتگی ہے شہید_گل_خزانی_شمع

جلے ہے دیکھ کے بالین_یار پر مجھ کو
نہ کیوں ہو دل پہ مرے داغ_بد_گمانی_شمع
