ہے داغ_عشق زینت_جیب_کفن ہنوز

ہے ناز_مفلساں زر_از_دست_رفتہ پر
ہوں گل_فروش_شوخی_داغ_کہن ہنوز

مے_خانۂ_جگر میں یہاں خاک بھی نہیں
خمیازہ کھینچے ہے بت_بے_داد_فن ہنوز

جوں جادہ سر_بہ_کوئے_تمناۓ_بے_دلی
زنجیر_پا ہے رشتۂ_حب_الوطن ہنوز
